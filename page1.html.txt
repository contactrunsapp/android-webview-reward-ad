<script>
let profiles = [];
let selectedPhotos = [];

function openJoinModal() {
  document.getElementById('joinModal').classList.add('active');
}

function closeJoinModal() {
  document.getElementById('joinModal').classList.remove('active');
}

function previewPhotos() {
  const files = document.getElementById('photoInput').files;
  const preview = document.getElementById('photoPreview');
  preview.innerHTML = '';
  selectedPhotos = [];
  
  for(let i = 0; i < files.length; i++) {
    const reader = new FileReader();
    reader.onload = function(e) {
      selectedPhotos.push(e.target.result);
      const img = document.createElement('img');
      img.src = e.target.result;
      preview.appendChild(img);
    };
    reader.readAsDataURL(files[i]);
  }
}

function loadProfiles() {
  const saved = localStorage.getItem('runshookup-profiles');
  if(saved) {
    profiles = JSON.parse(saved);
  } else {
    profiles = [
      {
        nickname: 'Sarah',
        age: '25-30',
        gender: 'Female',
        country: 'USA',
        city: 'New York',
        bio: 'Love hiking',
        photos: ['https://images.unsplash.com/photo-1494790108777-283128473ab8?w=400']
      },
      {
        nickname: 'Mike',
        age: '31-35',
        gender: 'Male',
        country: 'UK',
        city: 'London',
        bio: 'Food lover',
        photos: ['https://images.unsplash.com/photo-1500648767791-00dcc994a43e?w=400']
      }
    ];
    localStorage.setItem('runshookup-profiles', JSON.stringify(profiles));
  }
  displayProfiles();
}

function displayProfiles() {
  const container = document.getElementById('postsContainer');
  let html = '';
  
  profiles.forEach(p => {
    html += `
      <div class='profile-card'>
        <img class='profile-photo' src='${p.photos[0]}' onerror="this.src='https://via.placeholder.com/150/ff385c/fff?text=${p.nickname}'">
        <div class='profile-info'>
          <div class='profile-name'>${p.nickname}, ${p.age}</div>
          <div class='profile-location'>üìç ${p.city}</div>
          <div>
            <span class='profile-badge'>${p.country}</span>
            <span class='profile-badge'>${p.gender}</span>
          </div>
        </div>
      </div>
    `;
  });
  
  container.innerHTML = html;
}

function filterPosts() {
  const search = document.getElementById('searchFilter').value.toLowerCase();
  const country = document.getElementById('countryFilter').value;
  
  const cards = document.querySelectorAll('.profile-card');
  cards.forEach(card => {
    const name = card.querySelector('.profile-name').textContent.toLowerCase();
    const location = card.querySelector('.profile-location').textContent.toLowerCase();
    let show = true;
    
    if(search && !name.includes(search) && !location.includes(search)) show = false;
    if(country && !location.includes(country.toLowerCase())) show = false;
    
    card.style.display = show ? 'block' : 'none';
  });
}

function submitProfile() {
  const nickname = document.getElementById('nickname').value;
  const age = document.getElementById('age').value;
  const gender = document.getElementById('gender').value;
  const country = document.getElementById('joinCountry').value;
  const city = document.getElementById('joinCity').value;
  const bio = document.getElementById('bio').value;
  
  if(!nickname || !city || selectedPhotos.length === 0) {
    alert('Please fill required fields and add a photo');
    return;
  }
  
  const newProfile = {
    nickname,
    age,
    gender,
    country,
    city,
    bio,
    photos: selectedPhotos
  };
  
  profiles.push(newProfile);
  localStorage.setItem('runshookup-profiles', JSON.stringify(profiles));
  
  closeJoinModal();
  displayProfiles();
  alert('Profile created!');
}

// Populate country filter
const countries = ['USA', 'UK', 'Canada', 'Australia', 'India'];
const countryFilter = document.getElementById('countryFilter');
countries.forEach(c => {
  const option = document.createElement('option');
  option.value = c;
  option.textContent = c;
  countryFilter.appendChild(option);
});

// Initialize
loadProfiles();

// Close modal when tapping outside
window.addEventListener('click', (e) => {
  if(e.target.classList.contains('modal-overlay')) {
    closeJoinModal();
  }
});
